We'll use the mean step for all days to replace missing values in a new dataset

data2 <- data
na_ID <- which(is.na(data2$steps))
means <- rep(mean(data2$steps, na.rm = TRUE), times = length(na_ID))
data2[na_ID, "steps"] <- means
